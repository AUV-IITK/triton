86  // The measurement and its associated covariance
87  Eigen::VectorXd measurement_;
88  Eigen::MatrixXd covariance_;

90  // We want earlier times to have greater priority
91  bool operator()(const boost::shared_ptr<Measurement> &a, const boost::shared_ptr<Measurement> &b)
92  {
93  return (*this)(*(a.get()), *(b.get()));
94  }
95 
96  bool operator()(const Measurement &a, const Measurement &b)
97  {
98  return a.time_ > b.time_;
99  }

103  mahalanobisThresh_(std::numeric_limits<double>::max()),
104  time_(0.0),
105  topicName_("")

342  inline double computeControlAcceleration(const double state, const double control, const double accelerationLimit,
343  const double accelerationGain, const double decelerationLimit, const double decelerationGain)
344  {
345  FB_DEBUG("---------- FilterBase::computeControlAcceleration ----------\n");
346 
347  const double error = control - state;
348  const bool sameSign = (::fabs(error) <= ::fabs(control) + 0.01);
349  const double setPoint = (sameSign ? control : 0.0);
350  const bool decelerating = ::fabs(setPoint) < ::fabs(state);
351  double limit = accelerationLimit;
352  double gain = accelerationGain;
353 
354  if (decelerating)
355  {
356  limit = decelerationLimit;
357  gain = decelerationGain;
358  }
359 
360  const double finalAccel = std::min(std::max(gain * error, -limit), limit);
361 
362  FB_DEBUG("Control value: " << control << "\n" <<
363  "State value: " << state << "\n" <<
364  "Error: " << error << "\n" <<
365  "Same sign: " << (sameSign ? "true" : "false") << "\n" <<
366  "Set point: " << setPoint << "\n" <<
367  "Decelerating: " << (decelerating ? "true" : "false") << "\n" <<
368  "Limit: " << limit << "\n" <<
369  "Gain: " << gain << "\n" <<
370  "Final is " << finalAccel << "\n");
371 
372  return finalAccel;
373  }

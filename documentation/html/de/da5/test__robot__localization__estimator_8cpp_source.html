40 TEST(RLETest, StateBuffer)
41 {
42  // Generate a few estimator states
43  std::vector<RobotLocalization::EstimatorState> states;
44 
45  for ( int i = 0; i < 10; i++ )
46  {
47  /*
48  * t = i s;
49  * x = i m;
50  * vx = 1.0 m/s;
51  */
52  RobotLocalization::EstimatorState state;
53  state.time_stamp = i;
54  state.state(RobotLocalization::StateMemberX) = i;
55  state.state(RobotLocalization::StateMemberY) = 0;
56  state.state(RobotLocalization::StateMemberZ) = 0;
57 
58  state.state(RobotLocalization::StateMemberRoll) = 0;
59  state.state(RobotLocalization::StateMemberPitch) = 0;
60  state.state(RobotLocalization::StateMemberYaw) = 0;
61 
62  state.state(RobotLocalization::StateMemberVx) = 1;
63  state.state(RobotLocalization::StateMemberVy) = 0;
64  state.state(RobotLocalization::StateMemberVz) = 0;
65 
66  state.state(RobotLocalization::StateMemberVroll) = 0;
67  state.state(RobotLocalization::StateMemberVpitch) = 0;
68  state.state(RobotLocalization::StateMemberVyaw) = 0;
69 
70  state.state(RobotLocalization::StateMemberAx) = 0;
71  state.state(RobotLocalization::StateMemberAy) = 0;
72  state.state(RobotLocalization::StateMemberAz) = 0;
73  states.push_back(state);
74  }
75 
76  // Instantiate a robot localization estimator with a buffer capacity of 5
77  int buffer_capacity = 5;
78  Eigen::MatrixXd process_noise_covariance = Eigen::MatrixXd::Identity(RobotLocalization::STATE_SIZE,
79  RobotLocalization::STATE_SIZE);
80  RobotLocalization::RobotLocalizationEstimator estimator(buffer_capacity, RobotLocalization::FilterTypes::EKF,
81  process_noise_covariance);
82 
83  RobotLocalization::EstimatorState state;
84 
85  // Add the states in chronological order
86  for ( int i = 0; i < 6; i++ )
87  {
88  estimator.setState(states[i]);
89 
90  // Check that the state is added correctly
91  estimator.getState(states[i].time_stamp, state);
92  EXPECT_EQ(state.time_stamp, states[i].time_stamp);
93  }
94 
95  // We filled the buffer with more states that it can hold, so its size should now be equal to the capacity
96  EXPECT_EQ(static_cast<int>(estimator.getSize()), buffer_capacity);
97 
98  // Clear the buffer and check if it's really empty afterwards
99  estimator.clearBuffer();
100  EXPECT_EQ(estimator.getSize(), 0u);
101 
102  // Add states at time 1 through 3 inclusive to the buffer (buffer is not yet full)
103  for ( int i = 1; i < 4; i++ )
104  {
105  estimator.setState(states[i]);
106  }
107 
108  // Now add a state at time 0, but let's change it a bit (set StateMemberY=12) so that we can inspect if it is
109  // correctly added to the buffer.
110  RobotLocalization::EstimatorState state_2 = states[0];
111  state_2.state(RobotLocalization::StateMemberY) = 12;
112  estimator.setState(state_2);
113  EXPECT_EQ(RobotLocalization::EstimatorResults::Exact,
114  estimator.getState(states[0].time_stamp, state));
115 
116  // Check if the state is correctly added
117  EXPECT_EQ(state.state, state_2.state);
118 
119  // Add some more states. State at t=0 should now be dropped, so we should get the prediction, which means y=0
120  for ( int i = 5; i < 8; i++ )
121  {
122  estimator.setState(states[i]);
123  }
124  EXPECT_EQ(RobotLocalization::EstimatorResults::ExtrapolationIntoPast,
125  estimator.getState(states[0].time_stamp, state));
126  EXPECT_EQ(states[0].state, state.state);
127 
128  // Estimate a state that is not in the buffer, but can be determined by interpolation. The predicted state vector
129  // should be equal to the designed state at the requested time.
130  EXPECT_EQ(RobotLocalization::EstimatorResults::Interpolation,
131  estimator.getState(states[4].time_stamp, state));
132  EXPECT_EQ(states[4].state, state.state);
133 
134  // Estimate a state that is not in the buffer, but can be determined by extrapolation into the future. The predicted
135  // state vector should be equal to the designed state at the requested time.
136  EXPECT_EQ(RobotLocalization::EstimatorResults::ExtrapolationIntoFuture,
137  estimator.getState(states[8].time_stamp, state));
138  EXPECT_EQ(states[8].state, state.state);
139 
140  // Add missing state somewhere in the middle
141  estimator.setState(states[4]);
142 
143  // Overwrite state at t=3 (oldest state now in the buffer) and check if it's correctly overwritten.
144  state_2 = states[3];
145  state_2.state(RobotLocalization::StateMemberVy) = -1.0;
146  estimator.setState(state_2);
147  EXPECT_EQ(RobotLocalization::EstimatorResults::Exact,
148  estimator.getState(states[3].time_stamp, state));
149  EXPECT_EQ(state_2.state, state.state);
150 
151  // Add state that came too late
152  estimator.setState(states[0]);
153 
154  // Check if getState needed to do extrapolation into the past
155  EXPECT_EQ(estimator.getState(states[0].time_stamp, state),
156  RobotLocalization::EstimatorResults::ExtrapolationIntoPast);
157 
158  // Check state at t=0. This can only work correctly if the state at t=3 is
159  // overwritten and the state at zero is not in the buffer.
160  EXPECT_DOUBLE_EQ(3.0, state.state(RobotLocalization::StateMemberY));
161 }

166 TEST(FilterBaseTest, MeasurementProcessing)
167 {
168  using RobotLocalization::FilterDerived2;
169 
170  FilterDerived2 derived;
171 
172  Measurement meas;
173 
174  Eigen::VectorXd measurement(STATE_SIZE);
175  for (size_t i = 0; i < STATE_SIZE; ++i)
176  {
177  measurement[i] = 0.1 * static_cast<double>(i);
178  }
179 
180  Eigen::MatrixXd measurementCovariance(STATE_SIZE, STATE_SIZE);
181  for (size_t i = 0; i < STATE_SIZE; ++i)
182  {
183  for (size_t j = 0; j < STATE_SIZE; ++j)
184  {
185  measurementCovariance(i, j) = 0.1 * static_cast<double>(i * j);
186  }
187  }
188 
189  meas.topicName_ = "odomTest";
190  meas.measurement_ = measurement;
191  meas.covariance_ = measurementCovariance;
192  meas.updateVector_.resize(STATE_SIZE, true);
193  meas.time_ = 1000;
194 
195  // The filter shouldn't be initializedyet
196  EXPECT_FALSE(derived.getInitializedStatus());
197 
198  derived.processMeasurement(meas);
199 
200  // Now it's initialized, and the entire filter state
201  // should be equal to the first state
202  EXPECT_TRUE(derived.getInitializedStatus());
203  EXPECT_EQ(derived.getState(), measurement);
204 
205  // Process a measurement and make sure it updates the
206  // lastMeasurementTime variable
207  meas.time_ = 1002;
208  derived.processMeasurement(meas);
209  EXPECT_EQ(derived.getLastMeasurementTime(), meas.time_);
210 }

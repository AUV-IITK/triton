78  void FilterBase::reset()
79  {
80  initialized_ = false;
81 
82  // Clear the state and predicted state
83  state_.setZero();
84  predictedState_.setZero();
85  controlAcceleration_.setZero();
86 
87  // Prepare the invariant parts of the transfer
88  // function
89  transferFunction_.setIdentity();
90 
91  // Clear the Jacobian
92  transferFunctionJacobian_.setZero();
93 
94  // Set the estimate error covariance. We want our measurements
95  // to be accepted rapidly when the filter starts, so we should
96  // initialize the state's covariance with large values.
97  estimateErrorCovariance_.setIdentity();
98  estimateErrorCovariance_ *= 1e-9;
99 
100  // We need the identity for the update equations
101  identity_.setIdentity();
102 
103  // Set the epsilon matrix to be a matrix with small values on the diagonal
104  // It is used to maintain the positive-definite property of the covariance
105  covarianceEpsilon_.setIdentity();
106  covarianceEpsilon_ *= 0.001;
107 
108  // Assume 30Hz from sensor data (configurable)
109  sensorTimeout_ = 0.033333333;
110 
111  // Initialize our measurement time
112  lastMeasurementTime_ = 0;
113 
114  // These can be overridden via the launch parameters,
115  // but we need default values.
116  processNoiseCovariance_.setZero();
117  processNoiseCovariance_(StateMemberX, StateMemberX) = 0.05;
118  processNoiseCovariance_(StateMemberY, StateMemberY) = 0.05;
119  processNoiseCovariance_(StateMemberZ, StateMemberZ) = 0.06;
120  processNoiseCovariance_(StateMemberRoll, StateMemberRoll) = 0.03;
121  processNoiseCovariance_(StateMemberPitch, StateMemberPitch) = 0.03;
122  processNoiseCovariance_(StateMemberYaw, StateMemberYaw) = 0.06;
123  processNoiseCovariance_(StateMemberVx, StateMemberVx) = 0.025;
124  processNoiseCovariance_(StateMemberVy, StateMemberVy) = 0.025;
125  processNoiseCovariance_(StateMemberVz, StateMemberVz) = 0.04;
126  processNoiseCovariance_(StateMemberVroll, StateMemberVroll) = 0.01;
127  processNoiseCovariance_(StateMemberVpitch, StateMemberVpitch) = 0.01;
128  processNoiseCovariance_(StateMemberVyaw, StateMemberVyaw) = 0.02;
129  processNoiseCovariance_(StateMemberAx, StateMemberAx) = 0.01;
130  processNoiseCovariance_(StateMemberAy, StateMemberAy) = 0.01;
131  processNoiseCovariance_(StateMemberAz, StateMemberAz) = 0.015;
132 
133  dynamicProcessNoiseCovariance_ = processNoiseCovariance_;

136  void FilterBase::computeDynamicProcessNoiseCovariance(const Eigen::VectorXd &state, const double delta)
137  {
138  // A more principled approach would be to get the current velocity from the state, make a diagonal matrix from it,
139  // and then rotate it to be in the world frame (i.e., the same frame as the pose data). We could then use this
140  // rotated velocity matrix to scale the process noise covariance for the pose variables as
141  // rotatedVelocityMatrix * poseCovariance * rotatedVelocityMatrix'
142  // However, this presents trouble for robots that may incur rotational error as a result of linear motion (and
143  // vice-versa). Instead, we create a diagonal matrix whose diagonal values are the vector norm of the state's
144  // velocity. We use that to scale the process noise covariance.
145  Eigen::MatrixXd velocityMatrix(TWIST_SIZE, TWIST_SIZE);
146  velocityMatrix.setIdentity();
147  velocityMatrix.diagonal() *= state.segment(POSITION_V_OFFSET, TWIST_SIZE).norm();
148 
149  dynamicProcessNoiseCovariance_.block<TWIST_SIZE, TWIST_SIZE>(POSITION_OFFSET, POSITION_OFFSET) =
150  velocityMatrix *
151  processNoiseCovariance_.block<TWIST_SIZE, TWIST_SIZE>(POSITION_OFFSET, POSITION_OFFSET) *
152  velocityMatrix.transpose();
153  }

205  void FilterBase::processMeasurement(const Measurement &measurement)
206  {
207  FB_DEBUG("------ FilterBase::processMeasurement (" << measurement.topicName_ << ") ------\n");
208 
209  double delta = 0.0;
210 
211  // If we've had a previous reading, then go through the predict/update
212  // cycle. Otherwise, set our state and covariance to whatever we get
213  // from this measurement.
214  if (initialized_)
215  {
216  // Determine how much time has passed since our last measurement
217  delta = measurement.time_ - lastMeasurementTime_;
218 
219  FB_DEBUG("Filter is already initialized. Carrying out predict/correct loop...\n"
220  "Measurement time is " << std::setprecision(20) << measurement.time_ <<
221  ", last measurement time is " << lastMeasurementTime_ << ", delta is " << delta << "\n");
222 
223  // Only want to carry out a prediction if it's
224  // forward in time. Otherwise, just correct.
225  if (delta > 0)
226  {
227  validateDelta(delta);
228  predict(measurement.time_, delta);
229 
230  // Return this to the user
231  predictedState_ = state_;
232  }
233 
234  correct(measurement);
235  }
236  else
237  {
238  FB_DEBUG("First measurement. Initializing filter.\n");
239 
240  // Initialize the filter, but only with the values we're using
241  size_t measurementLength = measurement.updateVector_.size();
242  for (size_t i = 0; i < measurementLength; ++i)
243  {
244  state_[i] = (measurement.updateVector_[i] ? measurement.measurement_[i] : state_[i]);
245  }
246 
247  // Same for covariance
248  for (size_t i = 0; i < measurementLength; ++i)
249  {
250  for (size_t j = 0; j < measurementLength; ++j)
251  {
252  estimateErrorCovariance_(i, j) = (measurement.updateVector_[i] && measurement.updateVector_[j] ?
253  measurement.covariance_(i, j) :
254  estimateErrorCovariance_(i, j));
255  }
256  }
257 
258  initialized_ = true;
259  }
260 
261  if (delta >= 0.0)
262  {
263  lastMeasurementTime_ = measurement.time_;
264  }
265 
266  FB_DEBUG("------ /FilterBase::processMeasurement (" << measurement.topicName_ << ") ------\n");
267  }

351  void FilterBase::prepareControl(const double referenceTime, const double predictionDelta)
352  {
353  controlAcceleration_.setZero();
354 
355  if (useControl_)
356  {
357  bool timedOut = ::fabs(referenceTime - latestControlTime_) >= controlTimeout_;
358 
359  if (timedOut)
360  {
361  FB_DEBUG("Control timed out. Reference time was " << referenceTime << ", latest control time was " <<
362  latestControlTime_ << ", control timeout was " << controlTimeout_ << "\n");
363  }
364 
365  for (size_t controlInd = 0; controlInd < TWIST_SIZE; ++controlInd)
366  {
367  if (controlUpdateVector_[controlInd])
368  {
369  controlAcceleration_(controlInd) = computeControlAcceleration(state_(controlInd + POSITION_V_OFFSET),
370  (timedOut ? 0.0 : latestControl_(controlInd)), accelerationLimits_[controlInd],
371  accelerationGains_[controlInd], decelerationLimits_[controlInd], decelerationGains_[controlInd]);
372  }
373  }
374  }

384  bool FilterBase::checkMahalanobisThreshold(const Eigen::VectorXd &innovation,
385  const Eigen::MatrixXd &invCovariance,
386  const double nsigmas)
387  {
388  double sqMahalanobis = innovation.dot(invCovariance * innovation);
389  double threshold = nsigmas * nsigmas;
390 
391  if (sqMahalanobis >= threshold)
392  {
393  FB_DEBUG("Innovation mahalanobis distance test failed. Squared Mahalanobis is: " << sqMahalanobis << "\n" <<
394  "Threshold is: " << threshold << "\n" <<
395  "Innovation is: " << innovation << "\n" <<
396  "Innovation covariance is:\n" << invCovariance << "\n");
397 
398  return false;
399  }
400 
401  return true;
402  }

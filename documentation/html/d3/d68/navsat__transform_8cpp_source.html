215  void NavSatTransform::computeTransform()
216  {
217  // Only do this if:
218  // 1. We haven't computed the odom_frame->utm_frame transform before
219  // 2. We've received the data we need
220  if (!transform_good_ &&
221  has_transform_odom_ &&
222  has_transform_gps_ &&
223  has_transform_imu_)
224  {
225  // The UTM pose we have is given at the location of the GPS sensor on the robot. We need to get the UTM pose of
226  // the robot's origin.
227  tf2::Transform transform_utm_pose_corrected;
228  if (!use_manual_datum_)
229  {
230  getRobotOriginUtmPose(transform_utm_pose_, transform_utm_pose_corrected, ros::Time(0));
231  }
232  else
233  {
234  transform_utm_pose_corrected = transform_utm_pose_;
235  }
236 
237  // Get the IMU's current RPY values. Need the raw values (for yaw, anyway).
238  tf2::Matrix3x3 mat(transform_orientation_);
239 
240  // Convert to RPY
241  double imu_roll;
242  double imu_pitch;
243  double imu_yaw;
244  mat.getRPY(imu_roll, imu_pitch, imu_yaw);
245 
246  /* The IMU's heading was likely originally reported w.r.t. magnetic north.
247  * However, all the nodes in robot_localization assume that orientation data,
248  * including that reported by IMUs, is reported in an ENU frame, with a 0 yaw
249  * value being reported when facing east and increasing counter-clockwise (i.e.,
250  * towards north). To make the world frame ENU aligned, where X is east
251  * and Y is north, we have to take into account three additional considerations:
252  * 1. The IMU may have its non-ENU frame data transformed to ENU, but there's
253  * a possibility that its data has not been corrected for magnetic
254  * declination. We need to account for this. A positive magnetic
255  * declination is counter-clockwise in an ENU frame. Therefore, if
256  * we have a magnetic declination of N radians, then when the sensor
257  * is facing a heading of N, it reports 0. Therefore, we need to add
258  * the declination angle.
259  * 2. To account for any other offsets that may not be accounted for by the
260  * IMU driver or any interim processing node, we expose a yaw offset that
261  * lets users work with navsat_transform_node.
262  * 3. UTM grid isn't aligned with True East\North. To account for the difference
263  * we need to add meridian convergence angle.
264  */
265  imu_yaw += (magnetic_declination_ + yaw_offset_ + utm_meridian_convergence_);
266 
267  ROS_INFO_STREAM("Corrected for magnetic declination of " << std::fixed << magnetic_declination_ <<
268  ", user-specified offset of " << yaw_offset_ <<
269  " and meridian convergence of " << utm_meridian_convergence_ << "." <<
270  " Transform heading factor is now " << imu_yaw);
271 
272  // Convert to tf-friendly structures
273  tf2::Quaternion imu_quat;
274  imu_quat.setRPY(0.0, 0.0, imu_yaw);
275 
276  // The transform order will be orig_odom_pos * orig_utm_pos_inverse * cur_utm_pos.
277  // Doing it this way will allow us to cope with having non-zero odometry position
278  // when we get our first GPS message.
279  tf2::Transform utm_pose_with_orientation;
280  utm_pose_with_orientation.setOrigin(transform_utm_pose_corrected.getOrigin());
281  utm_pose_with_orientation.setRotation(imu_quat);
282 
283  utm_world_transform_.mult(transform_world_pose_, utm_pose_with_orientation.inverse());
284 
285  utm_world_trans_inverse_ = utm_world_transform_.inverse();
286 
287  ROS_INFO_STREAM("Transform world frame pose is: " << transform_world_pose_);
288  ROS_INFO_STREAM("World frame->utm transform is " << utm_world_transform_);
289 
290  transform_good_ = true;
291 
292  // Send out the (static) UTM transform in case anyone else would like to use it.
293  if (broadcast_utm_transform_)
294  {
295  geometry_msgs::TransformStamped utm_transform_stamped;
296  utm_transform_stamped.header.stamp = ros::Time::now();
297  utm_transform_stamped.header.frame_id = (broadcast_utm_transform_as_parent_frame_ ? "utm" : world_frame_id_);
298  utm_transform_stamped.child_frame_id = (broadcast_utm_transform_as_parent_frame_ ? world_frame_id_ : "utm");
299  utm_transform_stamped.transform = (broadcast_utm_transform_as_parent_frame_ ?
300  tf2::toMsg(utm_world_trans_inverse_) : tf2::toMsg(utm_world_transform_));
301  utm_transform_stamped.transform.translation.z = (zero_altitude_ ?
302  0.0 : utm_transform_stamped.transform.translation.z);
303  utm_broadcaster_.sendTransform(utm_transform_stamped);
304  }
305  }

308  bool NavSatTransform::datumCallback(robot_localization::SetDatum::Request& request,
309  robot_localization::SetDatum::Response&)
310  {
311  // If we get a service call with a manual datum, even if we already computed the transform using the robot's
312  // initial pose, then we want to assume that we are using a datum from now on, and we want other methods to
313  // not attempt to transform the values we are specifying here.
314  use_manual_datum_ = true;
315 
316  transform_good_ = false;
317 
318  sensor_msgs::NavSatFix *fix = new sensor_msgs::NavSatFix();
319  fix->latitude = request.geo_pose.position.latitude;
320  fix->longitude = request.geo_pose.position.longitude;
321  fix->altitude = request.geo_pose.position.altitude;
322  fix->header.stamp = ros::Time::now();
323  fix->position_covariance[0] = 0.1;
324  fix->position_covariance[4] = 0.1;
325  fix->position_covariance[8] = 0.1;
326  fix->position_covariance_type = sensor_msgs::NavSatStatus::STATUS_FIX;
327  sensor_msgs::NavSatFixConstPtr fix_ptr(fix);
328  setTransformGps(fix_ptr);
329 
330  nav_msgs::Odometry *odom = new nav_msgs::Odometry();
331  odom->pose.pose.orientation.x = 0;
332  odom->pose.pose.orientation.y = 0;
333  odom->pose.pose.orientation.z = 0;
334  odom->pose.pose.orientation.w = 1;
335  odom->pose.pose.position.x = 0;
336  odom->pose.pose.position.y = 0;
337  odom->pose.pose.position.z = 0;
338  odom->header.frame_id = world_frame_id_;
339  odom->child_frame_id = base_link_frame_id_;
340  nav_msgs::OdometryConstPtr odom_ptr(odom);
341  setTransformOdometry(odom_ptr);
342 
343  sensor_msgs::Imu *imu = new sensor_msgs::Imu();
344  imu->orientation = request.geo_pose.orientation;
345  imu->header.frame_id = base_link_frame_id_;
346  sensor_msgs::ImuConstPtr imu_ptr(imu);
347  imuCallback(imu_ptr);
348 
349  return true;
350  }

394  nav_msgs::Odometry NavSatTransform::utmToMap(const tf2::Transform& utm_pose) const
395  {
396  nav_msgs::Odometry gps_odom{};
397 
398  tf2::Transform transformed_utm_gps{};
399 
400  transformed_utm_gps.mult(utm_world_transform_, utm_pose);
401  transformed_utm_gps.setRotation(tf2::Quaternion::getIdentity());
402 
403  // Set header information stamp because we would like to know the robot's position at that timestamp
404  gps_odom.header.frame_id = world_frame_id_;
405  gps_odom.header.stamp = gps_update_time_;
406 
407  // Now fill out the message. Set the orientation to the identity.
408  tf2::toMsg(transformed_utm_gps, gps_odom.pose.pose);
409  gps_odom.pose.pose.position.z = (zero_altitude_ ? 0.0 : gps_odom.pose.pose.position.z);
410 
411  return gps_odom;
412  }

414  void NavSatTransform::mapToLL(const tf2::Vector3& point, double& latitude, double& longitude, double& altitude) const
415  {
416  tf2::Transform odom_as_utm{};
417 
418  tf2::Transform pose{};
419  pose.setOrigin(point);
420  pose.setRotation(tf2::Quaternion::getIdentity());
421 
422  odom_as_utm.mult(utm_world_trans_inverse_, pose);
423  odom_as_utm.setRotation(tf2::Quaternion::getIdentity());
424 
425  // Now convert the data back to lat/long and place into the message
426  NavsatConversions::UTMtoLL(odom_as_utm.getOrigin().getY(),
427  odom_as_utm.getOrigin().getX(),
428  utm_zone_,
429  latitude,
430  longitude);
431  altitude = odom_as_utm.getOrigin().getZ();
432  }

434  void NavSatTransform::getRobotOriginUtmPose(const tf2::Transform &gps_utm_pose,
435  tf2::Transform &robot_utm_pose,
436  const ros::Time &transform_time)
437  {
438  robot_utm_pose.setIdentity();
439 
440  // Get linear offset from origin for the GPS
441  tf2::Transform offset;
442  bool can_transform = RosFilterUtilities::lookupTransformSafe(tf_buffer_,
443  base_link_frame_id_,
444  gps_frame_id_,
445  transform_time,
446  ros::Duration(transform_timeout_),
447  offset);
448 
449  if (can_transform)
450  {
451  // Get the orientation we'll use for our UTM->world transform
452  tf2::Quaternion utm_orientation = transform_orientation_;
453  tf2::Matrix3x3 mat(utm_orientation);
454 
455  // Add the offsets
456  double roll;
457  double pitch;
458  double yaw;
459  mat.getRPY(roll, pitch, yaw);
460  yaw += (magnetic_declination_ + yaw_offset_ + utm_meridian_convergence_);
461  utm_orientation.setRPY(roll, pitch, yaw);
462 
463  // Rotate the GPS linear offset by the orientation
464  // Zero out the orientation, because the GPS orientation is meaningless, and if it's non-zero, it will make the
465  // the computation of robot_utm_pose erroneous.
466  offset.setOrigin(tf2::quatRotate(utm_orientation, offset.getOrigin()));
467  offset.setRotation(tf2::Quaternion::getIdentity());
468 
469  // Update the initial pose
470  robot_utm_pose = offset.inverse() * gps_utm_pose;
471  }
472  else
473  {
474  if (gps_frame_id_ != "")
475  {
476  ROS_WARN_STREAM_ONCE("Unable to obtain " << base_link_frame_id_ << "->" << gps_frame_id_ <<
477  " transform. Will assume navsat device is mounted at robot's origin");
478  }
479 
480  robot_utm_pose = gps_utm_pose;
481  }
482  }

484  void NavSatTransform::getRobotOriginWorldPose(const tf2::Transform &gps_odom_pose,
485  tf2::Transform &robot_odom_pose,
486  const ros::Time &transform_time)
487  {
488  robot_odom_pose.setIdentity();
489 
490  // Remove the offset from base_link
491  tf2::Transform gps_offset_rotated;
492  bool can_transform = RosFilterUtilities::lookupTransformSafe(tf_buffer_,
493  base_link_frame_id_,
494  gps_frame_id_,
495  transform_time,
496  transform_timeout_,
497  gps_offset_rotated);
498 
499  if (can_transform)
500  {
501  tf2::Transform robot_orientation;
502  can_transform = RosFilterUtilities::lookupTransformSafe(tf_buffer_,
503  world_frame_id_,
504  base_link_frame_id_,
505  transform_time,
506  transform_timeout_,
507  robot_orientation);
508 
509  if (can_transform)
510  {
511  // Zero out rotation because we don't care about the orientation of the
512  // GPS receiver relative to base_link
513  gps_offset_rotated.setOrigin(tf2::quatRotate(robot_orientation.getRotation(), gps_offset_rotated.getOrigin()));
514  gps_offset_rotated.setRotation(tf2::Quaternion::getIdentity());
515  robot_odom_pose = gps_offset_rotated.inverse() * gps_odom_pose;
516  }
517  else
518  {
519  ROS_WARN_STREAM_THROTTLE(5.0, "Could not obtain " << world_frame_id_ << "->" << base_link_frame_id_ <<
520  " transform. Will not remove offset of navsat device from robot's origin.");
521  }
522  }
523  else
524  {
525  ROS_WARN_STREAM_THROTTLE(5.0, "Could not obtain " << base_link_frame_id_ << "->" << gps_frame_id_ <<
526  " transform. Will not remove offset of navsat device from robot's origin.");
527  }
528  }

530  void NavSatTransform::gpsFixCallback(const sensor_msgs::NavSatFixConstPtr& msg)
531  {
532  gps_frame_id_ = msg->header.frame_id;
533 
534  if (gps_frame_id_.empty())
535  {
536  ROS_WARN_STREAM_ONCE("NavSatFix message has empty frame_id. Will assume navsat device is mounted at robot's "
537  "origin.");
538  }
539 
540  // Make sure the GPS data is usable
541  bool good_gps = (msg->status.status != sensor_msgs::NavSatStatus::STATUS_NO_FIX &&
542  !std::isnan(msg->altitude) &&
543  !std::isnan(msg->latitude) &&
544  !std::isnan(msg->longitude));
545 
546  if (good_gps)
547  {
548  // If we haven't computed the transform yet, then
549  // store this message as the initial GPS data to use
550  if (!transform_good_ && !use_manual_datum_)
551  {
552  setTransformGps(msg);
553  }
554 
555  double utmX = 0;
556  double utmY = 0;
557  std::string utm_zone_tmp;
558  NavsatConversions::LLtoUTM(msg->latitude, msg->longitude, utmY, utmX, utm_zone_tmp);
559  latest_utm_pose_.setOrigin(tf2::Vector3(utmX, utmY, msg->altitude));
560  latest_utm_covariance_.setZero();
561 
562  // Copy the measurement's covariance matrix so that we can rotate it later
563  for (size_t i = 0; i < POSITION_SIZE; i++)
564  {
565  for (size_t j = 0; j < POSITION_SIZE; j++)
566  {
567  latest_utm_covariance_(i, j) = msg->position_covariance[POSITION_SIZE * i + j];
568  }
569  }
570 
571  gps_update_time_ = msg->header.stamp;
572  gps_updated_ = true;
573  }
574  }

576  void NavSatTransform::imuCallback(const sensor_msgs::ImuConstPtr& msg)
577  {
578  // We need the baseLinkFrameId_ from the odometry message, so
579  // we need to wait until we receive it.
580  if (has_transform_odom_)
581  {
582  /* This method only gets called if we don't yet have the
583  * IMU data (the subscriber gets shut down once we compute
584  * the transform), so we can assumed that every IMU message
585  * that comes here is meant to be used for that purpose. */
586  tf2::fromMsg(msg->orientation, transform_orientation_);
587 
588  // Correct for the IMU's orientation w.r.t. base_link
589  tf2::Transform target_frame_trans;
590  bool can_transform = RosFilterUtilities::lookupTransformSafe(tf_buffer_,
591  base_link_frame_id_,
592  msg->header.frame_id,
593  msg->header.stamp,
594  transform_timeout_,
595  target_frame_trans);
596 
597  if (can_transform)
598  {
599  double roll_offset = 0;
600  double pitch_offset = 0;
601  double yaw_offset = 0;
602  double roll = 0;
603  double pitch = 0;
604  double yaw = 0;
605  RosFilterUtilities::quatToRPY(target_frame_trans.getRotation(), roll_offset, pitch_offset, yaw_offset);
606  RosFilterUtilities::quatToRPY(transform_orientation_, roll, pitch, yaw);
607 
608  ROS_DEBUG_STREAM("Initial orientation is " << transform_orientation_);
609 
610  // Apply the offset (making sure to bound them), and throw them in a vector
611  tf2::Vector3 rpy_angles(FilterUtilities::clampRotation(roll - roll_offset),
612  FilterUtilities::clampRotation(pitch - pitch_offset),
613  FilterUtilities::clampRotation(yaw - yaw_offset));
614 
615  // Now we need to rotate the roll and pitch by the yaw offset value.
616  // Imagine a case where an IMU is mounted facing sideways. In that case
617  // pitch for the IMU's world frame is roll for the robot.
618  tf2::Matrix3x3 mat;
619  mat.setRPY(0.0, 0.0, yaw_offset);
620  rpy_angles = mat * rpy_angles;
621  transform_orientation_.setRPY(rpy_angles.getX(), rpy_angles.getY(), rpy_angles.getZ());
622 
623  ROS_DEBUG_STREAM("Initial corrected orientation roll, pitch, yaw is (" <<
624  rpy_angles.getX() << ", " << rpy_angles.getY() << ", " << rpy_angles.getZ() << ")");
625 
626  has_transform_imu_ = true;
627  }
628  }
629  }

656  bool NavSatTransform::prepareFilteredGps(sensor_msgs::NavSatFix &filtered_gps)
657  {
658  bool new_data = false;
659 
660  if (transform_good_ && odom_updated_)
661  {
662  mapToLL(latest_world_pose_.getOrigin(), filtered_gps.latitude, filtered_gps.longitude, filtered_gps.altitude);
663 
664  // Rotate the covariance as well
665  tf2::Matrix3x3 rot(utm_world_trans_inverse_.getRotation());
666  Eigen::MatrixXd rot_6d(POSE_SIZE, POSE_SIZE);
667  rot_6d.setIdentity();
668 
669  for (size_t rInd = 0; rInd < POSITION_SIZE; ++rInd)
670  {
671  rot_6d(rInd, 0) = rot.getRow(rInd).getX();
672  rot_6d(rInd, 1) = rot.getRow(rInd).getY();
673  rot_6d(rInd, 2) = rot.getRow(rInd).getZ();
674  rot_6d(rInd+POSITION_SIZE, 3) = rot.getRow(rInd).getX();
675  rot_6d(rInd+POSITION_SIZE, 4) = rot.getRow(rInd).getY();
676  rot_6d(rInd+POSITION_SIZE, 5) = rot.getRow(rInd).getZ();
677  }
678 
679  // Rotate the covariance
680  latest_odom_covariance_ = rot_6d * latest_odom_covariance_.eval() * rot_6d.transpose();
681 
682  // Copy the measurement's covariance matrix back
683  for (size_t i = 0; i < POSITION_SIZE; i++)
684  {
685  for (size_t j = 0; j < POSITION_SIZE; j++)
686  {
687  filtered_gps.position_covariance[POSITION_SIZE * i + j] = latest_odom_covariance_(i, j);
688  }
689  }
690 
691  filtered_gps.position_covariance_type = sensor_msgs::NavSatFix::COVARIANCE_TYPE_KNOWN;
692  filtered_gps.status.status = sensor_msgs::NavSatStatus::STATUS_GBAS_FIX;
693  filtered_gps.header.frame_id = base_link_frame_id_;
694  filtered_gps.header.stamp = odom_update_time_;
695 
696  // Mark this GPS as used
697  odom_updated_ = false;
698  new_data = true;
699  }

701  return new_data;

704  bool NavSatTransform::prepareGpsOdometry(nav_msgs::Odometry &gps_odom)
705  {
706  bool new_data = false;
707 
708  if (transform_good_ && gps_updated_ && odom_updated_)
709  {
710  gps_odom = utmToMap(latest_utm_pose_);
711 
712  tf2::Transform transformed_utm_gps;
713  tf2::fromMsg(gps_odom.pose.pose, transformed_utm_gps);

718

720  tf2::Matrix3x3 rot(utm_world_transform_.getRotation());
721  Eigen::MatrixXd rot_6d(POSE_SIZE, POSE_SIZE);
722  rot_6d.setIdentity();
723

725  {
726  rot_6d(rInd, 0) = rot.getRow(rInd).getX();
727  rot_6d(rInd, 1) = rot.getRow(rInd).getY();
728  rot_6d(rInd, 2) = rot.getRow(rInd).getZ();
729  rot_6d(rInd+POSITION_SIZE, 3) = rot.getRow(rInd).getX();
730  rot_6d(rInd+POSITION_SIZE, 4) = rot.getRow(rInd).getY();
731  rot_6d(rInd+POSITION_SIZE, 5) = rot.getRow(rInd).getZ();
732  }

738  tf2::toMsg(transformed_utm_robot, gps_odom.pose.pose);
739  gps_odom.pose.pose.position.z = (zero_altitude_ ? 0.0 : gps_odom.pose.pose.position.z);

743  {

745  {

747  }
748  }

752  new_data = true;
